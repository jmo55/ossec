#This file contains my compiled notes from multiple resources to install and apply the initial configurations for OSSEC.

#Preferably be in root until completion, Update & Upgrade Distro

su

apt update && apt upgrade -y

#Prerequisite Commands;

apt install libz-dev libssl-dev libpcre2-dev build-essential libsystemd-dev git unzip apache2 php libapache2-mod-php postfix

#start/enable web service

systemctl start apache2

systemctl enable apache2

#INSTALL OSSEC

wget https://github.com/ossec/ossec-hids/archive/3.7.0.tar.gz

tar -xvzf 3.7.0.tar.gz

cd ossec-hids-3.7.0

./install.sh

:: answer the install questions - do a LOCAL & for email root@localhost

#Start ossec service

/var/ossec/bin/ossec-control start

#INSTALL OSSEC WEB UI

cd /tmp

git clone https://github.com/ossec/ossec-wui.git

mv /tmp/ossec-wui /var/www/html

cd /var/www/html/ossec-wui

chmod -R 755 /var/www/html/ossec-wui/

chown -R www-data:www-data /var/www/html/ossec-wui/

systemctl restart apache2

#CREATE AN AGENT | ADDING AGENT | EXTRACTING Key

/var/ossec/bin/manage_agents
